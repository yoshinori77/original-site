create table user_t(userID int NOT NULL PRIMARY KEY AUTO_INCREMENT, slackUserID varchar(255), name varchar(255),  password varchar(255), email varchar(255), sex int, birthday datetime, address varchar(255), newDate datetime);
create table chat_t(userID int NOT NULL PRIMARY KEY, slackUserID varchar(255), chat varchar(255), postTime varchar(255), profileImage varchar(255), numberOfLikes int, postFile varchar(255), FOREIGN KEY (userID) REFERENCES user_t(userID));
create table project_t(projectID int NOT NULL PRIMARY KEY AUTO_INCREMENT, projectName varchar(255), description varchar(255), projectImage varchar(255), newDate datetime, numberOfLikes int, postFile varchar(255));
