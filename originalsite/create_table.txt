create table user_t(userID int NOT NULL PRIMARY KEY AUTO_INCREMENT, slackUserID varchar(255) NOT NULL, userName varchar(255),  password varchar(255), sex int, birthday datetime, address varchar(255), skillID int, jobID int);
create table chat_t(userID int NOT NULL PRIMARY KEY, slackUserID varchar(255) NOT NULL, chat varchar(255), postTime varchar(255), profileImage varchar(255), numberOfLikes int, postFile varchar(255), FOREIGN KEY (userID) REFERENCES user_t(userID));
